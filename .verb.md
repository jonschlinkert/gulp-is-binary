## Example usage

```js
var gulp = require('gulp');
var isBinary = require('{%= name %}');

gulp.task('default', function() {
  return gulp.src('some/files/*.*')
    .pipe(isBinary())
    .pipe(through.obj(function(file, enc, next) {
      if (file.isNull()) {
        next(null, file);
        return;
      }
      console.log(file.isBinary());
      next(null, file);
    }))
});
```
